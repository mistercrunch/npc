unsigned long resolv(char *name) {
  unsigned long ip;
  struct hostent *he=NULL;
  ip=INADDR_NONE;

    he=gethostbyname (name);
    if (he != NULL) memcpy (&ip, he->h_addr, sizeof (ip));
  }
  return ip;
}

unsigned int conecta(struct sockaddr_in addr) {

  if((sockfd=socket(AF_INET,SOCK_STREAM,0))==-1)

  close(sockfd);
